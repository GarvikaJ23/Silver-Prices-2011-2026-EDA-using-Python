# Calculating daily returns (percentage change in closing price)
# Daily returns will show a key metric in finance to assess performance and risk.
# log returns are often preferred in financial modeling for their additive properties.
data['Daily_Return'] = np.log(data['Close'] / data['Close'].shift(1))

# Plotting the distribution of daily returns
plt.figure(figsize=(10, 6))
sns.histplot(data['Daily_Return'].dropna(), bins=50, kde=True, color='green')
plt.title('Distribution of Daily Returns', fontsize=16)
plt.xlabel('Daily Return', fontsize=12)
plt.ylabel('Frequency', fontsize=12)
plt.grid(True, alpha=0.7)
plt.tight_layout()
plt.show()

# Calculate and plot rolling volatility (e.g., 20-day rolling standard deviation of daily returns)
# Rolling volatility provides insight into how risk (price fluctuation) changes over time.
data['Rolling_Volatility'] = data['Daily_Return'].rolling(window=20).std() * np.sqrt(252) # Annualized volatility

plt.figure(figsize=(14, 7))
plt.plot(data['Rolling_Volatility'].dropna(), label='20-Day Annualized Rolling Volatility', color='red')

plt.ylabel('Annualized Volatility', fontsize=12)

# Calculate Simple Moving Averages (SMA)
# Short-term (e.g., 50-day) and long-term (e.g., 200-day) moving averages are common indicators of trend.
# Crossovers between these averages are often used by economists and traders to signal potential trend changes.
data['SMA_50'] = data['Close'].rolling(window=50).mean()
data['SMA_200'] = data['Close'].rolling(window=200).mean()

# Plotting Close price with 50-day and 200-day Moving Averages
plt.figure(figsize=(14, 7))
plt.plot(data['Close'], label='Silver Close Price', color='blue', alpha=0.7)
plt.plot(data['SMA_50'], label='50-Day SMA', color='orange')
plt.plot(data['SMA_200'], label='200-Day SMA', color='purple')
plt.title('Silver Closing Prices with Moving Averages', fontsize=16)
plt.xlabel('Date', fontsize=12)
plt.ylabel('Price (USD)', fontsize=12)
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.show()
